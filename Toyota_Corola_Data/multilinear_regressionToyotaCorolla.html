from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import OneHotEncoder, StandardScaler
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LinearRegression, Ridge, Lasso
from sklearn.metrics import r2_score

num = ['Age_08_04','KM','HP','cc','Doors','Gears','Quarterly_Tax','Weight']
cat = ['Fuel_Type','Automatic']

prep = ColumnTransformer([
    ('num', StandardScaler(), num),
    ('cat', OneHotEncoder(drop='first'), cat)
])

models = {
    'Linear': Pipeline([('prep', prep), ('lr', LinearRegression())]),
    'Ridge':  Pipeline([('prep', prep), ('ridge', Ridge(alpha=1.0))]),
    'Lasso':  Pipeline([('prep', prep), ('lasso', Lasso(alpha=0.001, max_iter=20000))]),
}

for name, pipe in models.items():
    pipe.fit(X_train, y_train)
    print(name, "R^2:", r2_score(y_test, pipe.predict(X_test)))
