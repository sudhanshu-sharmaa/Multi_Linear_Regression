from sklearn.preprocessing import OneHotEncoder, StandardScaler
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LinearRegression, Ridge, Lasso
from sklearn.metrics import r2_score

num = ['R&D Spend','Administration','Marketing Spend']
cat = ['State']

prep = ColumnTransformer([
    ('num','passthrough', num),
    ('cat', OneHotEncoder(drop='first', handle_unknown='ignore'), cat)
])

prep_scaled = ColumnTransformer([
    ('num', StandardScaler(), num),
    ('cat', OneHotEncoder(drop='first', handle_unknown='ignore'), cat)
])

models = {
    'Linear (one-hot)': Pipeline([('prep', prep), ('lr', LinearRegression())]),
    'Ridge (scaled)':   Pipeline([('prep', prep_scaled), ('ridge', Ridge(alpha=1.0))]),
    'Lasso (scaled)':   Pipeline([('prep', prep_scaled), ('lasso', Lasso(alpha=0.01, max_iter=10000))]),
}

r2_scores = {}
for name, pipe in models.items():
    pipe.fit(X_train, y_train)
    r2_scores[name] = r2_score(y_test, pipe.predict(X_test))

import pandas as pd
r2_df = pd.DataFrame({'Model': list(r2_scores.keys()), 'R^2': list(r2_scores.values())}).sort_values('R^2', ascending=False)
print(r2_df)
